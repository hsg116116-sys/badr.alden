          border-radius:12px;padding:10px 14px;margin-bottom:8px;
          display:flex;justify-content:space-between;align-items:center;color:#fff;
        }
        #badr-actions{padding:12px 24px 24px;display:flex;gap:12px;}
        #badr-accept{
          flex:1;height:56px;border-radius:16px;border:none;cursor:pointer;
          background:linear-gradient(135deg,#10b981,#059669);color:#fff;
          font-size:18px;font-weight:900;
          box-shadow:0 8px 24px rgba(16,185,129,0.4);
        }
        #badr-reject{
          width:120px;height:56px;border-radius:16px;cursor:pointer;
          background:rgba(239,68,68,0.12);color:#ef4444;
          border:1px solid rgba(239,68,68,0.25);font-size:15px;font-weight:700;
        }

      document.head.appendChild(style);
      let countdown = 60;
      let timer = null;
      let currentNotif = null;

        currentNotif = n;
        countdown = 60;
        document.getElementById('badr-notif-overlay')?.remove();
        const items = n.items || [];
        const overlay = document.createElement('div');
        overlay.id = 'badr-notif-overlay';
        overlay.innerHTML = `
          <div id="badr-notif-card">
            <div id="badr-progress-wrap" style="background:rgba(255,255,255,0.06)">
              <div id="badr-progress" style="width:100%"></div>
            </div>
            <div id="badr-header">
              <div style="display:flex;align-items:center;gap:14px">
                <div style="font-size:32px;background:rgba(16,185,129,0.2);border:2px solid #10b981;border-radius:14px;width:52px;height:52px;display:flex;align-items:center;justify-content:center">🛍️</div>
                <div>
                  <p style="color:#fff;font-size:20px;font-weight:900;margin:0">طلب جديد من الموقع!</p>
                  <p style="color:#34d399;font-size:13px;margin:4px 0 0">🟢 يُستلم بواسطة: اسلام</p>
                </div>
              </div>
              <div style="text-align:left">
                <p style="color:#10b981;font-size:30px;font-weight:900;margin:0">${Math.round(n.total)} <span style="font-size:14px">ج.م</span></p>
                <p id="badr-countdown" style="color:#6b7280;font-size:12px;margin:4px 0 0">ينتهي خلال 60ث</p>
              </div>
            </div>
            <div style="padding:14px 24px;display:grid;grid-template-columns:1fr 1fr;gap:10px">
              <div style="background:#1c1c1c;border:1px solid rgba(255,255,255,0.07);border-radius:12px;padding:10px 14px">
                <p style="color:#6b7280;font-size:10px;margin:0 0 4px;text-transform:uppercase">👤 العميل</p>
                <p style="color:#fff;font-size:13px;font-weight:600;margin:0">${n.customer_name || 'زبون'}</p>
              </div>
              <div style="background:#1c1c1c;border:1px solid rgba(255,255,255,0.07);border-radius:12px;padding:10px 14px">
                <p style="color:#6b7280;font-size:10px;margin:0 0 4px;text-transform:uppercase">📞 الهاتف</p>
                <p style="color:#fff;font-size:13px;font-weight:600;margin:0">${n.customer_phone || '—'}</p>
              </div>
              ${n.address ? `<div style="background:#1c1c1c;border:1px solid rgba(255,255,255,0.07);border-radius:12px;padding:10px 14px;grid-column:span 2">

                <p style="color:#fff;font-size:13px;font-weight:600;margin:0">${n.address}</p>
              </div>` : ''}
            </div>

              <p style="color:#9ca3af;font-size:11px;font-weight:700;margin:0 0 8px">المنتجات (${items.length} عنصر)</p>

                <div id="badr-item">
                  <div style="display:flex;align-items:center;gap:10px">
                    <span style="background:rgba(16,185,129,0.15);color:#10b981;border-radius:8px;padding:2px 8px;font-size:12px;font-weight:800">×${item.quantity}</span>
                    <span style="font-size:13px;font-weight:600">${item.name || item.productName || 'منتج'}</span>
                  </div>
                  <span style="color:#34d399;font-weight:800;font-size:13px">${Math.round((item.price||0)*(item.quantity||1))} ج.م</span>

              `).join('')}
            </div>
            <div id="badr-actions">
              <button id="badr-accept" onclick="window._badrAccept()">✅ استلام الطلب</button>
              <button id="badr-reject" onclick="window._badrReject()">❌ تجاهل</button>
            </div>
          </div>
        `;
        document.body.appendChild(overlay);
        // تشغيل صوت
        try {
          new Audio('https://assets.mixkit.co/active_storage/sfx/2869/2869-preview.mp3').play();
        } catch(e){}
        // عداد تنازلي
        clearInterval(timer);
        timer = setInterval(() => {
          countdown--;
          const el = document.getElementById('badr-countdown');
          const pr = document.getElementById('badr-progress');
          if (el) el.textContent = `ينتهي خلال ${countdown}ث`;
          if (pr) pr.style.width = `${(countdown/60)*100}%`;
          if (pr) pr.style.background = countdown > 30 ? '#10b981' : countdown > 15 ? '#f59e0b' : '#ef4444';
          if (countdown <= 0) { clearInterval(timer); hideNotification(); }
        }, 1000);
      }
      window._badrAccept = async function() {
        if (!currentNotif) return;
        document.getElementById('badr-accept').textContent = '⏳ جاري الاستلام...';
        await supabase.from('online_order_notifications').update({
          status: 'accepted', cashier_id: 1,
          accepted_by: 'اسلام', accepted_at: new Date().toISOString()
        }).eq('id', currentNotif.id);
        clearInterval(timer);
        hideNotification();
      };
      window._badrReject = function() {
        if (currentNotif) {
          supabase.from('online_order_notifications')
            .update({ status: 'rejected' }).eq('id', currentNotif.id);
        }
        clearInterval(timer);
        hideNotification();
      };
      function hideNotification() {
        const el = document.getElementById('badr-notif-overlay');
        if (el) el.remove();
        currentNotif = null;
      }
      // ── الاشتراك الفوري ────────────────────────────
      supabase.channel('badr_orders')
        .on('postgres_changes', {
          event: 'INSERT', schema: 'public',
          table: 'online_order_notifications'
        }, (payload) => {
          if (payload.new.status === 'pending') showNotification(payload.new);
        })
        .subscribe();
      console.log('✅ نظام إشعارات محمصة بدر الدين — جاهز');
    })();
